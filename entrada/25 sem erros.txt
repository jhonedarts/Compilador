funcao inteiro mergeSort(inteiro vetor<<<10>>>, inteiro posicaoInicio, inteiro posicaoFim)
inicio

    inicio
        metadeTamanho << (posicaoInicio + posicaoFim ) / 2;

        mergeSort(vetor, metadeTamanho + 1, posicaoFim);

        j << metadeTamanho + 1;

        inicio
            se (nao(i <> (metadeTamanho + 1) )) entao
            inicio
                vetorTemp<<<k>>> << vetor<<<j>>>;
                j << j + 1;
                k << k + 1;
            fim
            senao
            inicio
                se (nao(j <> (posicaoFim + 1))) entao
                inicio
                    vetorTemp<<<k>>> << vetor<<<i>>>;
                    j << j + 1;
                    k << k + 1;
                fim
                senao
                inicio
                    se (vetor<<<i>>> < vetor<<<j>>>) entao
                    inicio
                        vetorTemp<<<k>>> << vetor<<<i>>>;
                        j << j + 1;
                        k << k + 1;
                    fim
                    senao
                    inicio
                        vetorTemp<<<k>>> << vetor<<<j>>>;
                        j << j + 1;
                        k << k + 1;
                    fim
                fim
            fim
        fim

        enquanto ( i <= posicaoFim) faca
        inicio

            vetor<<<i>>> << vetorTemp<<<i-posicaoInicio>>>;

        fim
